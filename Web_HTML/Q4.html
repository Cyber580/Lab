STACKS USING ARRAY'S

            #include //stdio.h
            #define size 5
            int top=-1,s[5];
            void push(int);
            void pop();
            void traverse();
            void main()
            {
            int item,ch;
            while(1)
            {
            printf("\n Enter your choice 1.push 2.pop 3.traverse: ");
            scanf("%d",&ch);
            switch(ch)
            {
            case 1: printf("\n Enter the item: ");
            scanf("%d",&item);
            push(item);
            break;
            case 2: pop();
            break;
            case 3: traverse();
            break;
            default:exit(0);
            }
            }
            }
            void push(int item)
            {
            if(top==size - 1)
            {
            printf("\n Stack is full ");
            }
            else
            {
            top=top+1;
            s[top]=item;
            }
            }
            void pop()
            {
            if(top==-1)
            {
            printf("\n Stack is empty");
            }
            else
            {
            printf("Poped item is %d",s[top]);
            top=top-1;
            }
            }
            void traverse()
            {
            int i;
            if(top==-1)
            {
            printf("\n Stack is empty");
            }
            else
            {
            printf("\n Items of stack are: ");
            for(i=0;i<=top;i++)
            {
            printf("%d\t",s[i]);
            }
            }
            }
