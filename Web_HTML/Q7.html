            #include//stdio.h

            void enqueue();
            void dequeue();
            void travers();
            
            struct q
            {
             int ele;
             struct q *link;
            }*head,*ptr,*FRONT,*REAR,*new;
            
            void main()
            {
             int ch;
             clrscr();
             head = malloc(sizeof(struct q));
             head->link = NULL;
             head->ele = 0;
             FRONT = NULL;
             REAR = NULL;
             while(1)
             {
              printf("\n1)Enqueue\n2)denqueue\n3)travers\nenter choice:");
              scanf("%d",&ch);
              switch(ch)
              {
               case 1: enqueue();
                   break;
               case 2 : dequeue();
                    break;
               case 3 : travers();
                    break;
               default : exit(0);
            
              }
             }
            }
            
            void enqueue()
            {
             new = malloc(sizeof(struct q));
             new->link = NULL;
             printf("ENter element :");
             scanf("%d",&new->ele);
             if(FRONT ==NULL && REAR == NULL)
             {
              head->link = new;
              REAR = new;
              FRONT = new;
             }
             else
             {
              REAR->link =new;
              REAR =REAR->link;
            
             }
            }
            
            void dequeue()
            {
             if(FRONT == NULL && REAR == NULL)
             {
              printf("EMPTY Queue");
             }
              else if(REAR == FRONT)
              {
               printf("Dequeue item %d",FRONT->ele);
               FRONT = NULL;
               REAR = NULL;
              }
              else
              {
                head->link = FRONT->link;
                printf("Dequeue %d",FRONT->ele);
                free(FRONT);
                FRONT = head->link;
              }
            }
            
            void travers()
            {
             ptr = head;
             if(FRONT == NULL && REAR == NULL)
             {
              printf("EMPTY Queue");
             }
             else
             {
             printf("Queue elements are :\n");
             while(ptr->link!=NULL)
             {
              ptr = ptr->link;
              printf("%d\t",ptr->ele);
             }
             }
            }
