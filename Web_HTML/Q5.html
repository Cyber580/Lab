STACKS USING LINKED LIST'S

            #include //stdio.h
            #include //conio.h
            #include //stdlib
            
            struct s
            {
             int ele;
             struct s *link;
            }*head,*top;
            
            void push();
            void pop();
            void travers();
            
            void main()
            {
             int ch;
             head = malloc(sizeof(struct s));
             head->link = NULL;
             top = NULL;
             head->ele = 0;
             while(1)
             {
              printf("\n1)push\n2)pop\n3)travers\nenter Ur choice:");
              scanf("%d",&ch);
              switch(ch)
              {
               case 1 : push();
                    break;
               case 2 : pop();
                    break;
               case 3 : travers();
                    break;
               default : exit(0);
              }
             }
             
            }
            
            void push()
            {
             struct s *new = malloc(sizeof(struct s));
             new->link = NULL;
             printf("ENter element :");
             scanf("%d",&new->ele);
             new->link = top;
             head->link = new;
             top =  new;
            }
            
            void pop()
            {
             if(head->link == NULL)
             {
              printf("EMPTY Stack");
             }
             else
             {
              head->link = top->link;
              printf("Deleted item %d",top->ele);
              top=head->link;
             }
            }
            
            void travers()
            {
             struct s *ptr = head;
             if(head->link == NULL)
             {
              printf("EMPTY strack");
             }
             else
             {
              printf("Items in stack are :");
              while(ptr->link!=NULL)
              {
               ptr = ptr->link;
               printf("%d\t",ptr->ele);
              }
            
             }
            
            }
