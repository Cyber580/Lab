                      struct CLL *ptr = head->link;
                      head->link = ptr->link;
                      printf("DELETED IP from system %d",ptr->data);

                        struct CLL *ptr = head->link;

                            ptr1 = ptr;

                        }
                        ptr1->link = head;
                        printf("DELETED IP from system %d",ptr->data);
                        free(ptr);
                    }

                       struct CLL *ptr = head->link;
                       struct CLL *ptr1 , *ptr2;

                       {
                        ptr1 = ptr;
                        ptr = ptr->link;
                       }
                       if(ptr->link==head && ptr->data!=key)
                       {
                        printf("IP NOT FOUND");
                       }
                       else if(ptr->link==head && ptr->data==key)
                       {
                        ptr1->link = head;
                        printf("DELETED IP 28.0.0.%d",ptr->data);
                        free(ptr);
                       }
                       else
                       {
                        ptr2 = ptr->link;
                        ptr1->link = ptr2;  
                        printf("DELETED IP 28.0.0.%d",ptr->data);
                        free(ptr);
                       }
                    }
                  }
                }
                
                void traverse()
                {
                 if(head->link==head)
                 {
                    printf("\n!!ERROR!!,0 IP");

                 }
